def ProjWorkZone(workzone):
    """
    Projects the workzone at infile from inCRS coordinate system to outCRS coordinate system and saves it
    at the outfile position.
    workzone: workzone file
    inCRS: the coordinate system string of the input shapefile
    outCRS: the coordinate system string of the output shapefile
    return: None
    """
    transformer = Transformer.from_crs(crs_from='EPSG:4326',crs_to='EPSG:32129')

    # Transform beginning coordinates
    workzone['x_bgn'], workzone['y_bgn'] = transformer.transform(*zip(*workzone['LOCATION_LATLONG_BGN']))
    
    # Transform ending coordinates
    workzone['x_end'], workzone['y_end'] = transformer.transform(*zip(*workzone['LOCATION_LATLONG_END']))
    
    return workzone

def RoadwayProj(infile, outfile, outCRS):
    """
    Projects the shapefile at infile from inCRS coordinate system to outCRS coordinate system and saves it
    at the outfile position.
    infile: local or complete filename of file to transform
    outfile: local or complete filename of where to save the transformed shapefile
    outCRS: the coordinate system string of the output shapefile
    return: None
    """
    # If the infile is already the correct CRS, return
    with fiona.open(infile, 'r') as src:
        print("infile CRS:", src.crs)
        if src.crs == outCRS:
            print("Infile is correct CRS.")
            return
    
    # If the outfile already exists and is the correct CRS, return
    if os.path.exists(outfile):
        with fiona.open(outfile, 'r') as src:
            print("outfile CRS:", src.crs)
            if src.crs == outCRS:
                print("Outfile of correct CRS already exists.")
                return
    
    # Otherwise, transform and write to file
    all_roads = gpd.read_file(infile)
    all_roads.to_crs(outCRS, inplace=True)
    all_roads.to_file(outfile, driver='ESRI Shapefile')

def caldist_pl(x0, y0, line, index):
    """
    Calculates metadata about the relationship between a point and a line
    x0, y0: coordinates of the Point
    line: the line (as LineString)
    index: pass through parameter of the index on the road of that line segment
    """
    point = Point(x0, y0)

    # Calculate distance from the point to the line
    dist = point.distance(line)
    
    # Find the closest point on the line to point (x0, y0)
    closest_point = line.interpolate(line.project(point))

    # Extract the coordinates of the line's endpoints
    line_coords = list(line.coords)
    start_point = line_coords[0]
    end_point = line_coords[-1]

    return [index, closest_point, dist, [start_point, end_point]]

def direction_match(wzdiri, roaddiri):
    """
    Determines whether the direction of the workzone matches the direction of the road
    wzdiri: direction of the workzone
    roaddiri: direction of the road segment in question
    """
    if str(wzdiri)[0:1] == roaddiri:
        return True
    elif (str(wzdiri)[0:1] == 'B') and (roaddiri == 'O'):
        return True
    elif roaddiri == 'B':
        return True
    else:
        return False

def PointinBox1_vectorized(lon, lat, bufferlat, bufferlon, road):
    """
    Performs a vectorized operation to check if the points are within the bounding box
    """
    in_lon_range = (road['bbox_min_lon'] - bufferlon < lon) & (lon < road['bbox_max_lon'] + bufferlon)
    in_lat_range = (road['bbox_min_lat'] - bufferlat < lat) & (lat < road['bbox_max_lat'] + bufferlat)
    return in_lon_range & in_lat_range

def route_number_filter(workzone, roads):
    """
    Filters the roads down to the same route number as the workzone
    workzone: the workzone object
    roads: all roads to be filtered
    """
    # Create a zfilled version of 'wz_route' for comparison
    workzone['ST_RT_NO'] = workzone['ST_RT_NO'].astype(str).str.zfill(4)
    wz_route_set = set(workzone['ST_RT_NO'])
    road_strt_filtered = roads[roads['ST_RT_NO'].isin(wz_route_set)].copy()
    return road_strt_filtered

def direction_filter(workzone, roads):
    """
    Indicates whether the workzone and road is direction matched and places that boolean in another column
    Then filters only for rows in which the direction matches
    workzone: the workzone dataframe
    roads: the state roadway network dataframe
    returns the filtered dataframe
    """
    dir_f = roads['DIR_IND'].apply(lambda x: direction_match(workzone['DIRECTION'], x))
    roads['dir_filter'] = dir_f
    dir_filtered = roads[roads['dir_filter'] == True].copy()
    return dir_filtered

def geometry_filter(roads):
    """
    Filters out 'geometry' entries that are None, and pulls out bounds into their own columns for later ease
    roads: state road network dataframe
    returns: geometry_filtered, the filtered dataframe
    """
    # Remove rows that have no geometry
    geometry_filtered = roads[roads['geometry'] != None].copy()
    geometry_filtered = geometry_filtered[geometry_filtered['geometry'].type == 'LineString'].copy()
    

    # Pull out the bounding box values into their own columns
    bounds_df = geometry_filtered['geometry'].apply(lambda x: pd.Series(x.bounds, index=['bbox_min_lon', 'bbox_min_lat', \
                                                                                         'bbox_max_lon', 'bbox_max_lat']))
    geometry_filtered = pd.concat([geometry_filtered, bounds_df], axis=1)

    return geometry_filtered

def traverse_segments(road, start, end, start_with_1):
    """
    Traverses from either start or end (depending on if start_with_1 is True)
    and traverses along 'connectivity' of road segments until the other index is reached
    Returns the entire trimmed road of the workzone
    road: the road segments to snap to
    start: the start data of the workzone
    end: the end data of the workzone
    start_with_1: start with start index if True, start with end index if False
    """
    p1idx = start['road_idx']
    p2idx = end['road_idx']
    segments = []
    if start_with_1:
        i = p1idx
    else:
        i = p2idx

    if i in road.index:
        r = road.loc[i]
    else:
        return segments, False, False, pd.DataFrame()

    found_start = False
    found_end = False
    result_road = road.copy()
    while (not found_start or not found_end):
        if i == p1idx and i == p2idx:
            found_start = True
            found_end = True
            result_road.at[p1idx, 'geometry'] = \
                trim_two_points(r.loc['geometry'], start['closest_point'], end['closest_point'])
        elif i == p1idx:
            found_start = True
            result_road.at[p1idx, 'geometry'] = \
                trim(r.loc['geometry'], start['closest_point'], (not found_end))
        elif i == p2idx:
            found_end = True
            result_road.at[p2idx, 'geometry'] = \
                trim(r.loc['geometry'], end['closest_point'], (not found_start))

        if (found_start and not found_end) or (found_end and not found_start):
            segments.append(i)

        if found_end and found_start:
            segments.append(i)
            break

        if r['connected_to'] != []:
            i = r['connected_to'][0]
            if i in result_road.index:
                r = result_road.loc[i].copy()
            else:
                break
        else:
            break
    
    return segments, found_start, found_end, result_road

def get_road_segments(road, start, end):
    """
    Returns all of the road segments encapsulated in the workzone
    road: all road segments to sort through
    start: start of the workzone
    end: end of the workzone
    """
    pd.options.mode.chained_assignment = None
    success = True

    if 'connected_to' not in road.columns:
        road['first_coord'] = road['geometry'].apply(lambda x: x.coords[0])
        road['last_coord'] = road['geometry'].apply(lambda x: x.coords[-1])
        road['connected_to'] = road.apply(lambda x: road.index[road['first_coord'] == x['last_coord']].tolist(), axis=1)

    segments, found_start, found_end, result_road = traverse_segments(road, start, end, True)
    
    if not found_end or not found_start:
         segments, found_start, found_end, result_road = traverse_segments(road, start, end, False)

    if not found_end or not found_start:
        success = False

    pd.options.mode.chained_assignment = 'warn'
    return result_road.loc[segments], success

def trim(line, point, is_start, start_is_end=False):
    """
    Trims the line according to the point
    line: the line to snap to the point and trim
    point: the point to trim according to
    is_start: whether it is the start or not -- determines which side of the line to keep
    """

    # Find the nearest point on the line to the given point
    dists = [Point(x).distance(point) for x in line.coords[:]]
    dists = pd.DataFrame(data=dists, index=range(0,(len(dists))))

    dists.sort_values(by=0,ascending=True, inplace=True)
    top_two_closest = dists.head(2)
    
    geoms = split(snap(line, point, float(0.01)), point)
        
    nearest_vertex_index = top_two_closest.head().index[0]
    
    if is_start:
        if len(geoms.geoms) == 1:
            if nearest_vertex_index == 0:
                return geoms.geoms[0]
            else:
                return LineString()
        else:
            return geoms.geoms[1]
    else:
        if len(geoms.geoms) == 1:
            if nearest_vertex_index != 0:
                return geoms.geoms[0]
            else:
                return LineString()
        else:
            return geoms.geoms[0]
        
def trim_two_points(line, point1, point2):
    """
    Special case of "trim" which must be called when both the end and start point of the workzone are on the SAME segment.
    line: the line segment to trim
    point1: the first point to trim excess from
    point2: the second point from which to trim the excess off the line

    return: the final trimmed line
    """
    snapped_line = snap(line, point1, 0.01)
    snapped_line = snap(snapped_line, point2, 0.01)
    
    fragments = split(snapped_line, point1)
    
    for fragment in fragments.geoms:
        if point2.distance(fragment) == 0: 
            final_fragments = split(fragment, point2)
            for final_fragment in final_fragments.geoms:
                if point1.distance(final_fragment) == 0 and point2.distance(final_fragment) == 0:
                    return final_fragment
    return None

def min_wz_roadway_points(workzone, roads, bufferlon = 10, bufferlat = 10):
    """
    Map matches workzones onto the state roadway network.
    workzone: the workzone dataframe
    roads: the state roadway network
    returns start_segment_df, end_df which contain the map matched workzone starting points and ending points respectively
    """
    # Remove irrelevant roads from the road network
    roads_st_rt_filtered = route_number_filter(workzone, roads)

    # pull the geometry bounds out into their own columns and remove invalid geometries
    geometry_filtered = geometry_filter(roads_st_rt_filtered)

    start_df = pd.DataFrame()
    end_df = pd.DataFrame()
    workzone_segments = pd.DataFrame()
    for i, wz in workzone.iterrows():
        # Filter dir_filtered based on the bounding box condition for start and end points
        st_rt_no = wz['ST_RT_NO'].zfill(4)

        # For each workzone, only look at those that share the same state route number
        st_rt_filtered = geometry_filtered[geometry_filtered['ST_RT_NO'] == st_rt_no].copy()

        # For each workzone, only look at those that share the same direction as this workzone
        dir_filtered = st_rt_filtered[(st_rt_filtered['DIR_IND'] == wz['DIRECTION'][0:1]) |
                                      (st_rt_filtered['DIR_IND'] == 'B') |
                                      ((st_rt_filtered['DIR_IND'] == 'O') & (wz['DIRECTION'][0:1] == 'B'))]

        # Grab the points on the road network that are within the bounds of the workzone
        start_in_box = PointinBox1_vectorized(wz['x_bgn'], wz['y_bgn'], bufferlat, bufferlon, dir_filtered)
        end_in_box = PointinBox1_vectorized(wz['x_end'], wz['y_end'], bufferlat, bufferlon, dir_filtered)

        # Starting points: Narrow the field of view to only those points within the bounding box
        filtered_start = dir_filtered[start_in_box].copy()
        if len(filtered_start) > 0:
            results = filtered_start.apply(lambda x: caldist_pl(wz["x_bgn"], wz['y_bgn'], x['geometry'], x.name), axis=1)
            sresultsdf = pd.DataFrame({
                'road_idx': results.apply(lambda x: x[0]),
                'workzone_idx': wz.name,
                'x_bgn': wz["x_bgn"],
                'y_bgn': wz["y_bgn"],
                'closest_point': results.apply(lambda x: x[1]),
                'dist': results.apply(lambda x: x[2]),
                'bounds': results.apply(lambda x: x[3])
            })
            start_df = pd.concat([start_df, sresultsdf])

        # Ending points: Narrow the field of view to only those points within the bounding box
        filtered_end = dir_filtered[end_in_box].copy()
        if len(filtered_end) > 0:
            results = filtered_end.apply(lambda x: caldist_pl(wz["x_end"], wz['y_end'], \
                                                              x['geometry'], x.name), axis=1)
            eresultsdf = pd.DataFrame({
                'road_idx': results.apply(lambda x: x[0]),
                'workzone_idx': wz.name,
                'x_end': wz["x_end"],
                'y_end': wz["y_end"],
                'closest_point': results.apply(lambda x: x[1]),
                'dist': results.apply(lambda x: x[2]),
                'bounds': results.apply(lambda x: x[3])
            })
            end_df = pd.concat([end_df, eresultsdf])

        segments, success = get_road_segments(dir_filtered, sresultsdf.loc[sresultsdf['dist'].idxmin()], \
                                     eresultsdf.loc[eresultsdf['dist'].idxmin()])
        
        if (not segments.empty) and success:
            segments = segments[['geometry']]
            multi_index = pd.MultiIndex.from_arrays([[i] * len(segments), segments.index], names=('Workzone_Index', 'Road_Index'))
            temp_segments = segments.copy()  
            temp_segments.index = multi_index
            workzone_segments = pd.concat([workzone_segments, temp_segments])

    return start_df, end_df, workzone_segments, roads
